JUnit单元测试

  1. 测试分类

    1. 黑盒测试：不需要代码，输入值，检查程序输出
    2. 白盒测试：需要代码，关注程序执行流程

  2. JUnit白盒测试

    1. 步骤：

      1. 定义一个测试类（测试用例）（类名为…Test）（包名为….test）
      2. 定义测试方法（可以独立运行(不需要调用)）（方法名为test(方法名)（返回值为void）（参数为空）
      3. 给方法加注解@Test(能独立运行)：需要导入JUnti依赖环境
      4. 断言结果(assert):Assert.assertEquals(5,result);

    2. 具体代码：

    3. @before和@after

      1. @before用于初始化对象方法:
        @before
        public void init(){}
      2. @after用于释放资源方法:
        @after
        public void close(){}

反射

  1. 反射机制：将类的各个组成部分封装成其他对象
  2. 先理解Java代码实现的三个阶段：
    1. 源代码(Source)阶段：javac编译”类.java”文件，生成字节码”类.class”字节码文件，其中包括：
      1. 成员变量
      2. 构造函数
      3. 成员方法
    2. Class类对象阶段：
      1. 类加载器ClassLoader读取”类.class”字节码文件
      2. Class类对象进行封装：
        1. 成员变量:Field[] fields;
        2. 构造函数:Constructor[] constructors;
        3. 成员方法:Method[] methods;
      3. 当执行new Class-name()时创建对象
    3. 运行时(Runtime)阶段:new Class()生成类对象
  3. 反射用处：
    1. 在程序运行时操作封装的对象.（如调用方法这一过程就是一个反射）
    2. 解耦

1. Class对象

  1. 获取Class对象的方式
    1. Class.forName(“类名”)：将字节码文件加载进内存。（配置文件）
    2. 类名.class:通过类名的属性class获取。（参数传递）
    3. 对象.getClass():getClass()方法在Object类中定义(即所有对象都有该方法)。（对象获取字节码文件）
    4. 总结：同一个字节码文件(类.class)在一次程序运行过程中只会加载一次，获取一个Class对象

public class ReflectTest1 {
public static void main(String[] args) throws Exception {
//获取Person字节码文件对象(3种方式)
//1. Class.forName(“类名”)
Class person1 = Class.forName(“Reflect反射.Person”); //forName静态方法,默认在src路径下,返回一个Class对象
System.out.println(person1);

    //2. 类名.class
    Class person2 = Person.class;
    System.out.println(person2);

    //3. 对象.getClass()
    Person per = new Person();
    Class person3 = per.getClass();
    System.out.println(person3);

    //比较三个对象
    System.out.println(person1==person2);    //"=="基本数据类型：比较值;对象类型：比较内存地址
    System.out.println(person1.equals(person2));    //"equals"基本数据类型：判断值;重写后看实际情况
    System.out.println(person1==person3);
    System.out.println(person1.equals(person3));
}

}

import java.lang.reflect.Constructor;
import java.lang.reflect.Field;
import java.lang.reflect.Method;

public class ReflectTest2 {
public static void main(String[] args) throws Exception {
/*
1. 获取成员变量
2. 获取构造函数
3. 获取成员方法
*/
//0. 获取Class对象,实例化类对象
Class person1 = Person.class; //获取Class类对象
Person p = new Person(); //实例化类对象

    //1. 获取成员变量
    //1.1 获取所有的public成员变量:getFields()
    Field[] fields1 = person1.getFields();      //获取所有的public成员变量
    for (Field field : fields1) {
        System.out.println(field);      //迭代输出所有的public变量
    }
    //1.2 获取指定的public成员变量
    Field fields2 = person1.getField("height");      //获取指定名称的public成员变量
    //1.3 设置public成员变量的值：set
    fields2.set(p,170);
    //1.4 获取public成员变量的值：get
    Object get_public = fields2.get(p);
    System.out.println(get_public);

    //1.5 获取全部成员变量:getDeclaredFields(),忽略访问权限
    Field[] fields3 = person1.getDeclaredFields();      //获取所有成员变量
    for (Field field : fields3) {
        System.out.println(field);
    }
    //1.6 获取指定成员变量：
    Field field4 = person1.getDeclaredField("age");         //private
    Field field5 = person1.getDeclaredField("height");      //public
    //1.7.1 设置私有成员变量的值
    //1.7.1.1 忽略访问权限
    field4.setAccessible(true);    //暴力反射
    field4.set(p,20);
    //1.7.2 获取私有成员变量的值
    Object get_age = field4.get(p);
    System.out.println(get_age);
    //1.7.3 设置public成员变量的值：set
    field5.set(p,170);
    //1.7.4 获取public成员变量的值：get
    Object get_height = field5.get(p);
    System.out.println(get_height);


    //2.获取构造函数对象
    Constructor constructor1 = person1.getConstructor();    //获取默认的构造函数
    Constructor constructor2 = person1.getConstructor(String.class,int.class);      //获取参数为String和int类型的构造函数
    //2.1 使用构造函数对象,创建对象：newInstance
    Object object1 = constructor1.newInstance();     //空参构造函数
    System.out.println(object1);
    Object object2 = constructor2.newInstance("李四",20); //有参数的构造函数
    System.out.println(object2);
    System.out.println("--------------------");


    //3. 获取成员方法对象:方式一：指定方法名；方式二：指定参数
    Method method1 = person1.getMethod("meth1");    //public无参数方法
    Method method2 = person1.getDeclaredMethod("meth2", String.class);        //private无参数方法
    //3.1 使用成员方法对象,执行方法：invoke(method,values)
    method1.invoke(p);          //执行空参数方法
    method2.setAccessible(true);        //private方法暴力反射
    method2.invoke(p,"execute");        //执行String参数方法
}

}

    import java.io.InputStream;
    import java.lang.reflect.Constructor;
    import java.lang.reflect.Method;
    import java.util.Properties;

/*
可以创建任意类的对象，并可以执行任意方法
*/
public class ReflectTest3 {
public static void main(String[] args) throws Exception {
//1. 获取配置文件路径
ClassLoader classLoader = ReflectTest3.class.getClassLoader();
InputStream inputStream = classLoader.getResourceAsStream(“Reflect.properties”);
//2. 加载配置文件：
Properties pro = new Properties();
pro.load(inputStream);
//3. 获取配置文件参数
String classname = pro.getProperty(“className”);
String methodname1 = pro.getProperty(“methodName1”);
String methodname2 = pro.getProperty(“methodName2”);
//4. 加载类进内存:生成字节码文件,方便Class对象获取成员
Class cls = Class.forName(classname);
//5. 获取构造函数对象
Constructor constructor = cls.getConstructor();
//6. 创建对象
Object object1 = constructor.newInstance();
//7. 获取方法对象
Method method1 = cls.getMethod(methodname1); //获取名为methodname1的方法
Method method2 = cls.getDeclaredMethod(methodname2,String.class); //获取名为methodname2的方法,其参数类型为String
method2.setAccessible(true);
//8. 执行方法：
method1.invoke(object1);
method2.invoke(object1,”执行private”); //传入参数，调用对应的方法
}
}

  1. 注解
    package Annotation;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

/*
使用注解替代配置文件
*/
@Target(value = {ElementType.TYPE})
@Retention(RetentionPolicy.RUNTIME)
public @interface PropertiesAnno {
//1. 类名
String classname();
//2. 方法名
String methodname();
}

@PropertiesAnno(classname = “Annotation.Person”,methodname = “meth2”) //设置注解属性
public class ReflectTest3 {
public static void main(String[] args) throws Exception {
//1. 解析注解
//1.1 获取该类的字节码文件对象，返回一个参数类型为本类的Class对象
Class reflectTest3Class = ReflectTest3.class;
//1.2 获取注解对象:getAnnotation(注解.class)
PropertiesAnno annotation = reflectTest3Class.getAnnotation(PropertiesAnno.class);
//1.3 注解接口引用对象获取返回的属性值
String classname = annotation.classname();
String methodname = annotation.methodname();

Class cls = Class.forName(classname);
//1.5 获取构造函数对象
Constructor constructor = cls.getConstructor();
//1.6 创建对象
Object object = constructor.newInstance();
//1.7 获取方法对象
Method method = cls.getDeclaredMethod(methodname);
method.setAccessible(true);
//1.8 执行方法
method.invoke(object);
}
}

    import java.io.BufferedWriter;
    import java.io.FileWriter;
    import java.io.IOException;
    import java.lang.reflect.Method;

public class Test {
public static void main(String[] args) throws IOException {
//1. 创建测试方法类对象
Meths meths = new Meths();
//2. 获取字节码文件
Class cls = meths.getClass();
//3. 获取所有方法
Method[] methods = cls.getDeclaredMethods();
//4. 遍历方法,判断方法是否含义@Check注解
int number = 0; //异常次数
BufferedWriter bw = new BufferedWriter(new FileWriter(“异常清单.txt”)); //新建文件”异常处理”来皴法异常日记
for (Method method : methods) {
if (method.isAnnotationPresent(Check.class)) {
try {
//有则执行
method.setAccessible(true);
method.invoke(meths);
} catch (Exception e) {
//在catch中捕获异常
number++;
bw.write(method+”出现异常.”);
bw.newLine(); //光标进入下一行
bw.write(“异常名称:”+e.getCause().getClass().getSimpleName()); //获取异常名称
bw.newLine();
bw.write(“异常原因”+e.getCause().getMessage());
bw.newLine();
bw.write(“—————-“);
bw.newLine();
}
}
}
bw.write(“本次测试一共出现”+number+”次异常.”);
bw.flush();
bw.close();
}
}
